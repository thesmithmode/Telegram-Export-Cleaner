/**
* Telegram HTML-safe Chunker v2.1
* • Работает с parse_mode = HTML
* • Экранирует любые теги, кроме whitelisted
* • Опционально конвертирует Markdown-таблицы
* • Делит сообщение на куски ≤ CHUNK_SIZE
*/

/* ───── Параметры ───── */
const RAW_TEXT = typeof $json.output === 'string' ? $json.output : '';
const CHUNK_SIZE = +($env.CHUNK_SIZE ?? 4000);
const ENABLE_TABLE = ($env.ENABLE_TABLE ?? 'true') === 'true';
const ALLOWED_TAGS = ['b','i','u','s','code','pre','a']; // whitelist

/* ───── Helpers ───── */

// базовое экранирование
const escapeHTML = s => s
.replace(/&/g,'&amp;')
.replace(/</g,'&lt;')
.replace(/>/g,'&gt;')
.replace(/"/g,'&quot;')
.replace(/'/g,'&#39;');

// разрешаем только whitelisted-теги
function whitelistTags(str){
const placeholders = [];
// 1) временно прячем разрешённые теги
const masked = str.replace(/<\/?([a-z0-9]+)(\b[^>]*)?>/gi,(m,tag)=>{
if(ALLOWED_TAGS.includes(tag.toLowerCase())){
const id = placeholders.push(m) - 1;
return §§${id}§§;
}
return m; // остальные теги будут экранированы
});
// 2) глобально экранируем
let safe = escapeHTML(masked);
// 3) возвращаем whitelisted-теги
placeholders.forEach((tag,i)=>{
safe = safe.replace(§§${i}§§, tag);
});
return safe;
}

// Markdown-таблица → HTML-список
function pipeTableToList(md){
const lines = md.split('\n').filter(l=>/^\|.*\|$/.test(l));
if (lines.length < 2) return md;
const hdr = lines[0].split('|').map(t=>t.trim()).filter(Boolean);
const rows = lines.slice(2).map(r=>r.split('|').map(c=>c.trim()).filter(Boolean));
return rows.map(r =>
r.map((c,i)=><b>${escapeHTML(hdr[i])}</b>: ${escapeHTML(c)}).join('\n')
).join('\n\n');
}

// делим текст и «переносим» открытые теги
function splitSmartHTML(src){
const out = [];
let openTags = [];
for(let s=0; s<src.length; ){
let e = s + CHUNK_SIZE;
if(e < src.length){
const br = Math.max(src.lastIndexOf('\n',e), src.lastIndexOf('. ',e), src.lastIndexOf(' ',e));
if(br > s + 200) e = br;
}
let chunk = src.slice(s,e);

// какие теги открылись / закрылись в куске
const open = [...chunk.matchAll(/<([a-z]+)(?: [^>]*?)?>/gi)].map(m=>m[1]);
const close = [...chunk.matchAll(/<\/([a-z]+)>/gi)].map(m=>m[1]);

// добавляем недозакрытые теги с прошлого куска
if(openTags.length){
chunk = openTags.map(t=><${t}>).join('') + chunk;
}

// обновляем стек
open.forEach(t => openTags.push(t));
close.forEach(t => {
const idx = openTags.lastIndexOf(t);
if(idx > -1) openTags.splice(idx,1);
});

// закрываем теги в конце куска
if(openTags.length){
chunk += openTags.slice().reverse().map(t=></${t}>).join('');
}

out.push(chunk.trim());
s = e;
}
return out;
}

/* ───── Pipeline ───── */
let txt = RAW_TEXT;

// 1) конвертируем таблицы (если нужно)
if(ENABLE_TABLE && /\|.*\|/.test(txt)){
txt = txt.replace(/((?:\|.+\n?)+)/g, m => pipeTableToList(m));
}

// 2) убираем zero-width и control-символы
txt = txt.replace(/[\u200B-\u200D\uFEFF]/g,'')
.replace(/[\x00-\x09\x0B-\x1F]/g,'');

// 3) NFC-нормализация
txt = txt.normalize('NFC');

// 4) экранируем + whitelist
txt = whitelistTags(txt);

// 5) делим и отдаём
return splitSmartHTML(txt).map(chunk => ({ json: { message: chunk } }));

Привет! Если совсем просто, AgentKit — это способ дать ИИ не только болтать, но и делать дела.

Агент — это “мозг”: понимает, что ты написал, сам решает, что сделать и в каком порядке.

Инструменты/экшены — это его “руки”: сходить в веб (поиск), прочитать файлы, запустить код, дернуть твой API, даже “покликать” компьютер. Ты сам решаешь, какие руки дать.

Память/контекст — чтобы не забывать, о чём вы говорили и из каких файлов тянули факты.

Handoff — когда одному мозгу тесно, он зовёт другого агента-эксперта.

Гардрейлы и трейсинг — безопасность и “чёрный ящик”: видно, какие шаги агент сделал, с какими параметрами, где ошибся.

Agent Builder / ChatKit — собрать агента без боли и сразу воткнуть чат-виджет на сайт/продукт.

Evals — авто-проверки качества: где агент тупит, где окей, чтобы быстро улучшать.

Проще думай так: ты описываешь цель, AgentKit даёт ИИ план + кнопки для реальных действий. Если кнопок мало — докручиваешь свои API как новые “умения”.

Ключевые шаги для вашего workflow в n8n
1. Получите доступ к API Wildberries

Зайдите в личный кабинет продавца на Wildberries.

В настройках профиля найдите раздел «API-интеграции».

Создайте новый ключ API. Обязательно выдайте ему права для категории «Цены и скидки», это критично для обновления цен.

Помните, что ключ действует 180 дней, потом его нужно обновить.

2. Основы работы с API Wildberries

API работает по стандарту HTTP и возвращает данные в формате JSON.

Есть лимиты на количество запросов. Например, может быть ограничение в 300 запросов в минуту. Если превысить лимит, сервер вернет ошибку 429. В этом случае нужно сделать паузу.

🔶 Промпт-инжиниринг в ноябре 2025: что реально работает

Сегодня копался в свежих исследованиях по промптингу. Нашел три техники, которые выводят взаимодействие с LLM на принципиально другой уровень. Коротко о главном.

━━━━━━━━━━━━━━━

✔️ Verbalized Sampling
*решение проблемы mode collapse*

Вышла работа исследователей в 2025. Суть простая: модели выдают только топовый ответ, игнорируя альтернативы. Это называется mode collapse.

Новый подход: вместо обычного запроса просишь модель дать 5 вариантов с указанием вероятности каждого.

→ Результат: модель раскрывает весь спектр решений. Особенно заметно на творческих задачах.

🔗 [Verbalized Sampling на Hugging Face](https://huggingface.co/papers/2409.20449)

━━━━━━━━━━━━━━━━

✔️ Graph of Thoughts уничтожает Chain-of-Thought

Фундаментальное исследование. Цифры впечатляют по сравнению с традиционным CoT.

Логика простая: вместо линейной цепочки рассуждений используется граф с ветвлениями. Для сложных задач это принципиально другой уровень.

🔗 [Graph of Thoughts - arXiv](https://arxiv.org/abs/2308.09687)

━━━━━━━━━━━━━━━━

✔️ Парадокс reasoning-моделей

Тут самое интересное. Исследования показали, что для o1 и o3-mini сложные техники промптинга работают хуже простых.

Раньше писал:
*"Давай решим это пошагово, сначала определим переменные..."*

Теперь пишу:
*"Решите эту задачу."*

→ Модель умеет рассуждать сама. Не нужно её учить.

🔗 [Prompting o1 - VentureBeat](https://venturebeat.com/ai/how-to-prompt-on-openais-new-o1-models/)

━━━━━━━━━━━━━━━━

✔️ Sketch-of-Thought
*экономия токенов*

Метод сжимает рассуждения на 84% без потери качества. Вместо многословного Chain-of-Thought используются когнитивно-сжатые конструкции.

🔗 [Sketch-of-Thought - arXiv](https://arxiv.org/abs/2503.05179)

━━━━━━━━━━━━━━━

➤ Что делать:

1. Тестируйте Verbalized Sampling на задачах, где нужны варианты

2. Упростите промпты для o1. Удалите все "думай пошагово" и подобное

3. Для сложных многошаговых задач переходите на граф-структуры

### Google ломает парадигму обучения ИИ с помощью Nested Learning и архитектуры HOPE

В сообществе исследователей ИИ давно назревала проблема, ставшая ахиллесовой пятой большинства нейросетей — катастрофическое забывание. Модели, обучаясь новым задачам, безвозвратно теряли навыки, приобретенные ранее. Google Research, похоже, нашла фундаментальное решение, представив новую парадигму Nested Learning (вложенное обучение) и доказательство её концепции — архитектуру HOPE.[2][4]

Суть Nested Learning заключается в отказе от монолитного процесса обучения. Вместо этого модель рассматривается как иерархия вложенных задач оптимизации, каждая из которых обновляется со своей частотой. На верхних, "медленных" уровнях аккумулируются фундаментальные, устойчивые знания. На нижних, "быстрых" — небольшие модули, которые оперативно переобучаются для усвоения новой информации, не затрагивая ядро знаний. Сами оптимизаторы в этой парадигме трактуются как форма ассоциативной памяти, которая помнит, какие изменения и как вносились ранее.[6][2]

На основе этого принципа была создана HOPE — самомодифицирующаяся рекуррентная архитектура, являющаяся развитием идей, заложенных в модели Titans. Ключевое отличие HOPE — это Continuum Memory System (CMS). Если у традиционных трансформеров есть два жёстких блока памяти (краткосрочная в механизме внимания и долгосрочная в весах), то у HOPE это целый спектр модулей, обновляющихся с разной скоростью, что создаёт гибкую и многоуровневую систему памяти. Это позволяет архитектуре иметь практически неограниченное количество уровней контекстного обучения.[3][8]

Результаты экспериментов с моделью HOPE 1.5B впечатляют:
* Непрерывное обучение: Модель последовательно обучилась пяти новым задачам, не показав деградации в выполнении предыдущих. И это без использования стандартных техник вроде replay или регуляризации.[3]
* Языковое моделирование: HOPE достигла SOTA-показателей по точности (перплексии) для своего размера, опережая многие более крупные модели.[3]
* Длинный контекст: Архитектура эффективно работает с контекстом до 128 тысяч токенов, демонстрируя превосходство над современными трансформерами в тестах "Needle-In-Haystack" (поиск информации в больших объёмах текста).[3]

Эта технология открывает дорогу к созданию по-настоящему адаптивных ИИ-агентов, способных обучаться в процессе диалога с пользователем без необходимости полного и дорогостоящего переобучения. Хотя HOPE пока является прототипом, её успех может ознаменовать поворотный момент в развитии ИИ, приближая нас к системам с непрерывным жизненным циклом обучения.[5][6]
